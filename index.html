<script>

  const todayKey = new Date().toISOString().slice(0, 10);


  document.getElementById('date').textContent =

    new Date().toLocaleDateString(undefined, { weekday: 'long', month: 'short', day: 'numeric' });


  function loadHabits() {

    return JSON.parse(localStorage.getItem('habits') || '[]');

  }


  function saveHabits(habits) {

    localStorage.setItem('habits', JSON.stringify(habits));

  }


  function render() {

    const list = document.getElementById('habitList');

    const habits = loadHabits();

    list.innerHTML = '';


    habits.forEach((habit, index) => {

      const doneToday = habit.doneDates.includes(todayKey);


      const row = document.createElement('div');

      row.className = 'habit';


      row.innerHTML = `

        <div class="habit-name">${habit.name}</div>

        <div class="habit-actions">

          <div class="check ${doneToday ? 'done' : ''}" onclick="toggleDone(${index})">

            ${doneToday ? '✓' : ''}

          </div>

          <button class="delete" onclick="deleteHabit(${index})">×</button>

        </div>

      `;


      list.appendChild(row);

    });

  }


  function addHabit() {

    const input = document.getElementById('habitInput');

    const name = input.value.trim();

    if (!name) return;


    const habits = loadHabits();

    habits.push({ name, doneDates: [] });

    saveHabits(habits);


    input.value = '';

    render();

  }


  function toggleDone(index) {

    const habits = loadHabits();

    const dates = habits[index].doneDates;


    const i = dates.indexOf(todayKey);

    if (i >= 0) dates.splice(i, 1);

    else dates.push(todayKey);


    saveHabits(habits);

    render();

  }


  function deleteHabit(index) {

    const habits = loadHabits();

    habits.splice(index, 1);

    saveHabits(habits);

    render();

  }


  render();

</script>
